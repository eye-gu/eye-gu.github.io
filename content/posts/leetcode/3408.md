https://leetcode.cn/problems/design-task-manager/description/?envType=daily-question&envId=2025-09-18

依照昨天延迟删除的思路, 在优先级队列中的任务poll出来后, 判断是否存在, 优先级是否一致.

```java
class TaskManager {

    Map<Integer, int[]> taskMapping = new HashMap<>();
    PriorityQueue<int[]> taskQueue = new PriorityQueue<>((a, b) -> {
        if (a[1] == b[1]) {
            return b[0] - a[0];
        }
        return b[1] - a[1];
    });

    public TaskManager(List<List<Integer>> tasks) {
        for (List<Integer> task : tasks) {
            add(task.get(0), task.get(1), task.get(2));
        }
    }

    public void add(int userId, int taskId, int priority) {
        taskMapping.put(taskId, new int[]{userId, priority});
        taskQueue.offer(new int[]{taskId, priority});
    }

    public void edit(int taskId, int newPriority) {
        int[] ints = taskMapping.get(taskId);
        if (ints == null) {
            return;
        }
        ints[1] = newPriority;
        taskQueue.offer(new int[]{taskId, newPriority});
    }

    public void rmv(int taskId) {
        taskMapping.remove(taskId);
    }

    public int execTop() {
        while (!taskQueue.isEmpty()) {
            int[] task = taskQueue.poll();
            int[] ints = taskMapping.get(task[0]);
            if (ints == null) {
                continue;
            }
            if (ints[1] != task[1]) {
                continue;
            }
            taskMapping.remove(task[0]);
            return ints[0];
        }
        return -1;
    }
}
```
