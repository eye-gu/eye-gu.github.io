https://leetcode.cn/problems/smallest-subarrays-with-maximum-bitwise-or/description/

```java
class Solution {
        public int[] smallestSubarrays(int[] nums) {
            int xor = 0;
            int n = nums.length;
            // rightMaxXor[i] 表示 [i,n)的最大异或值
            int[] rightMaxXor = new int[n];
            for (int i = n - 1; i >= 0; i--) {
                xor |= nums[i];
                rightMaxXor[i] = xor;
            }
            // [i,right)可以等于最大异或值
            int right = 1;
            // 存储每一位1的个数
            int[] positionCount = new int[31];
            for (int j=0; j<31; j++) {
                if ((nums[0] & (1 << j)) != 0) {
                    positionCount[j]++;
                }
            }
            int[] res = new int[n];
            for (int i = 0; i < n; i++) {
                // 为0,说明后面的都是0
                if (rightMaxXor[i] == 0) {
                    res[i] = 1;
                    continue;
                }
                
                // 剔除掉上一个数的1
                if (i > 0) {
                    for (int j=0; j<31; j++) {
                        if ((nums[i-1] & (1 << j)) != 0) {
                            positionCount[j]--;
                        }
                    }
                }
                // 构造当前的异或值
                xor = 0;
                for (int j = 0; j < 31; j++) {
                    if (positionCount[j] > 0) {
                        xor |= (1 << j);
                    }
                }
                // 小于最大的异或值,right指针向右移动
                while (right < n && xor < rightMaxXor[i]) {
                    for (int j=0; j<31; j++) {
                        if ((nums[right] & (1 << j)) != 0) {
                            if (positionCount[j] == 0) {
                                xor |= (1 << j);
                            }
                            positionCount[j]++;
                        }
                    }
                    right++;
                }
                res[i] = right - i;
            }
            return res;
        }
    }
```

官解的思路

```go
func smallestSubarrays(nums []int) []int {
	n := len(nums)
  // pos[i]记录二进制的i位上为1, 最小的nums数组的下标
	pos := make([]int, 31)
	ans := make([]int, n)
	for i := n - 1; i >= 0; i-- {
		k := i
		for j := 0; j < 31; j++ {
			if nums[i]&(1<<j) != 0 {
        // 不等于0说明j位上为1
				pos[j] = i
			} else {
        // nums[i]在j位上为0, pos[j]不等于0说明在j位上,右侧有过1
				if pos[j] != 0 {
					k = max(k, pos[j])
				}
			}
		}
		ans[i] = k - i + 1
	}
	return ans
}
```
