https://leetcode.cn/problems/count-substrings-that-satisfy-k-constraint-ii/description



```java
class Solution {
    public long[] countKConstraintSubstrings(String s, int k, int[][] queries) {
        int n = s.length();
        int[] count = new int[2];
        // right[i]表示 [i, right[i])是符合k约束的最大数据, 即[i, right[i]]就不符合k约束
        int[] right = new int[n];
        // 初始值为n, 因为数组右侧不会被赋值, 默认可以到尾部
        Arrays.fill(right, n);
        // prefix[i]表示[0, i]范围内满足k约束的子数组个数
        long[] prefix = new long[n + 1];
        for (int i = 0, j = 0; j < n; ++j) {
            count[s.charAt(j) - '0']++;
            while (count[0] > k && count[1] > k) {
                count[s.charAt(i) - '0']--;
                right[i] = j;
                i++;
            }
            // prefix[i]表示[0, i]范围内满足k约束的子数组个数
            // 所以 prefix[j + 1] 和 prefix[j]的差别就是以j+1结尾的满足k约束的子数组个数
            // i是目前满足k约束的左端点, 所以以j+1结尾的满足k约束的子数组个数就等于数组长度 j-i+1
            prefix[j + 1] = prefix[j] + j - i + 1;
        }


        long[] res = new long[queries.length];
        for (int q = 0; q < queries.length; q++) {
            int l = queries[q][0], r = queries[q][1];
            int i = Math.min(right[l], r + 1);
          	// 因为[l,i)满足k约束, 所以所有子数据都满足
            // [l,i)的长度为i - l, 设为n, 所有子数组分为长度1的n个,长度2的(n-1)个...长度n的1个, 就是递增公式
            long part1 = (long) (i - l + 1) * (i - l) / 2;
            // 剩下部分是[i,r], 满足k约束,右端点在[i,r]之间,左端点必然大于l, 因此可以通过前缀和相减
            long part2 = prefix[r + 1] - prefix[i];
            res[q] = part1 + part2;
        }
        return res;
    }
}
```
