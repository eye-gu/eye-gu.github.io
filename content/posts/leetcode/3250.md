```rust
impl Solution {
    pub fn count_of_pairs(nums: Vec<i32>) -> i32 {
        let &max = nums.iter().max().unwrap();
        let max = max as usize;
        let n = nums.len();
        let MOD = 1000000007;
        // dp[i][j] 表示arr1[i]的值为j时, 数组的数量
        let mut dp = vec![vec![0; max + 1]; n];

        // 第一个位置上, 只要arr1[0] >= 0 && arr1[0] <= nums[0], 都可以
        for i in 0..=nums[0] as usize {
            dp[0][i] = 1;
        }

        for i in 1..n {
            // 当前位置上可以的取值范围
            for v in 0..=nums[i] {
                // 上一个位置的取值范围, 一定要小于当前位置
                for v_last in 0..=v {
                    // 满足上个位置的arr2大于当前位置的arr2
                    if nums[i - 1] - v_last >= nums[i] - v {
                        dp[i][v as usize] = (dp[i][v as usize] + dp[i - 1][v_last as usize]) % MOD;
                    }
                }
            }
        }

        let mut res = 0;
        for &v in &dp[n - 1] {
            res = (res + v ) % MOD;
        }
        res
    }
}
```
