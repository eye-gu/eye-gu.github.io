https://leetcode.cn/problems/taking-maximum-energy-from-the-mystic-dungeon/description/?envType=daily-question&envId=2025-10-10



```java
class Solution {
    public int maximumEnergy(int[] energy, int k) {
        int n = energy.length;
        int ans = Integer.MIN_VALUE;
        // 倒序遍历. 只需遍历k次
        for (int i=n-1; i>=n-k; i--) {
            int energySum = 0;
            // 以i结尾的序列, 倒序遍历, 计算能量
            for (int j=i; j>=0; j-=k) {
                energySum += energy[j];
                ans = Math.max(ans, energySum);
            }
        }
        return ans;
    }
}
```
