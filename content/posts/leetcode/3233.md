https://leetcode.cn/problems/find-the-count-of-numbers-which-are-not-special/description



```java
class Solution {
    public int nonSpecialCount(int l, int r) {
      	// 满足题目条件的只能是平方数, 且是素数的平方
        int n = (int) Math.sqrt(r);
        int[] v = new int[n + 1];
        int res = r - l + 1;
        for (int i = 2; i <= n; i++) {
            if (v[i] == 0) {
                if (i * i >= l && i * i <= r) {
                    res--;
                }
              	// 埃氏筛
              	// i一开始为2, 然后将范围内所有2的倍数划去
              	// 然后为3
              	// 然后为5(4已经被划去)
              	// 剩下的第一个没有被划去的数就为下一个素数
                for (int j = i * 2; j <= n; j += i) {
                    v[j] = 1;
                }
            }
        }
        return res;
    }
}
```
