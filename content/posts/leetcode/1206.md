https://leetcode.cn/problems/design-skiplist/description/



了解了跳表的实现. 之前虽然了解跳表的数据结构, 还以为是通过两个指针(同层下一个节点, 下一层相同数据节点)实现的.

通过一个概率来随机一个层数, 很神奇.

```go
// maxLevel 定义跳表的最大层数
const maxLevel = 32

// p 定义随机层数时节点提升一层的概率
const p = 0.25

type SkiplistNode struct {
	val int
	// next[i]代表第i层下一个节点
	// 注意0层代表最底层
	next []*SkiplistNode
}

type Skiplist struct {
	// -1, 头节点
	head *SkiplistNode
	// 当前的层数
	level int
}

func Constructor() Skiplist {
	return Skiplist{&SkiplistNode{-1, make([]*SkiplistNode, maxLevel)}, 0}
}

func (this *Skiplist) Search(target int) bool {
	cur := this.head
	// 从最高层开始逐层向下搜索
	for i := this.level - 1; i >= 0; i-- {
		// 在当前层找到小于目标值的最大节点
		for cur.next[i] != nil && cur.next[i].val < target {
			cur = cur.next[i]
		}
	}
	// 移动到最底层的下一个节点
	// 前面找的都是小于target的节点, 最底层的下一个节点就是第一个大于等于target的节点
	cur = cur.next[0]
	return cur != nil && cur.val == target
}

func (this *Skiplist) randomLevel() int {
	level := 1
	for level < maxLevel && rand.Float64() < p {
		level++
	}
	return level
}

func (this *Skiplist) Add(num int) {
	update := make([]*SkiplistNode, maxLevel)
	for i := 0; i < maxLevel; i++ {
		update[i] = this.head
	}
	cur := this.head
	for i := this.level - 1; i >= 0; i-- {
		for cur.next[i] != nil && cur.next[i].val < num {
			cur = cur.next[i]
		}
		// update存储当前层小于num的最大节点, 即需要将num插入update[i]后面, update[i]后面的节点插入到num后面
		update[i] = cur
	}
	lv := this.randomLevel()
	this.level = max(lv, this.level)
	newNode := &SkiplistNode{num, make([]*SkiplistNode, lv)}
	for i := 0; i < lv; i++ {
		// 新节点的下一个节点指向当前层更新节点的下一个节点
		newNode.next[i] = update[i].next[i]
		// 当前层更新节点的下一个节点指向新节点
		update[i].next[i] = newNode
	}
}

func (this *Skiplist) Erase(num int) bool {
	update := make([]*SkiplistNode, maxLevel)
	for i := 0; i < maxLevel; i++ {
		update[i] = this.head
	}
	cur := this.head
	for i := this.level - 1; i >= 0; i-- {
		// 在当前层级上移动，直到找到第一个大于或等于目标元素的节点的前一个
		for cur.next[i] != nil && cur.next[i].val < num {
			cur = cur.next[i]
		}
		// update存储当前层小于num的最大节点, 即后面一个节点如果是num，则需要删除
		update[i] = cur
	}
	cur = cur.next[0]
	if cur == nil || cur.val != num {
		return false
	}
	// 从最底层开始, 如果update[i]的下一个节点是num，则需要删除
	for i := 0; i < this.level && update[i].next[i] == cur; i++ {
		// 对第 i 层的状态进行更新，将 forward 指向被删除节点的下一跳
		update[i].next[i] = cur.next[i]
	}
	// 如果删除节点后，某些层级变得空了，那么更新当前的 level
	for this.level > 1 && this.head.next[this.level-1] == nil {
		this.level--
	}
	return true
}
```
