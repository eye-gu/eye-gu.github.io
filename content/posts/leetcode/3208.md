```rust
impl Solution {
    pub fn number_of_alternating_groups(colors: Vec<i32>, k: i32) -> i32 {
        let n = colors.len();
        // 开始的坐标
        let mut start = 0;
        let mut res = 0;
        // 以start-1坐标开头, 到start + k - 2, 是否是交替组
        let mut success = false;
        let k = k as usize;
        while start < n {
            if success {
                // 如果start-1是交替组, 那么只要校验末尾的两个颜色是否不同
                if colors[(start + k - 1) % n] != colors[(start + k - 2) % n] {
                    start += 1;
                    res += 1;
                } else {
                    // 如果末尾两个颜色相同, start可以直接跳到末尾, 中间必然不会成为交替组
                    success = false;
                    start = start + k - 1;
                }
            } else {
                let mut flag = true;
                // 从start遍历
                for j in 1..k {
                    // 如果中间出现颜色相同, 那么start可以直接跳到不同的地方
                    if colors[(start + j) % n] == colors[(start + j -  1) % n] {
                        start += j;
                        flag = false;
                        break;
                    }
                }

                if flag {
                    success = true;
                    res += 1;
                    start += 1;
                }
            }
        }

        res
    }
}
```

```rust
impl Solution {
    pub fn number_of_alternating_groups(colors: Vec<i32>, k: i32) -> i32 {
        let n = colors.len() as i32;
        let mut res = 0;
        // 左右不同的数量, 初始为1是因为k个交替组, 只需要k-1个连续不同
        let mut cnt = 1;
        for i in (-k + 2)..n {
            // 左右不同就加1, 如果相同, 那么cnt变为初始值, 代表需要从该位置重新计算
            if colors[((i + n) % n) as usize] != colors[((i - 1 + n) % n) as usize] {
                cnt += 1;
            } else {
                cnt = 1;
            }
            if cnt >= k {
                res += 1;
            }
        }
        res
    }
}
```
