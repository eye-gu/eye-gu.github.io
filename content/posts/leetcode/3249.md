```rust
impl Solution {
    pub fn count_good_nodes(edges: Vec<Vec<i32>>) -> i32 {
        let n = edges.len() + 1;
				// 构建邻接表
        let mut edge_vec = vec![vec![]; n];
        for edge in edges {
            let (a, b) = (edge[0], edge[1]);
            edge_vec[a as usize].push(b);
            edge_vec[b as usize].push(a);
        }
				// tree_count[i] 表示i为根节点的树的节点数量
        let mut tree_count = vec![0; n];
      	// child_vec[i] 表示i的直接子节点
        let mut child_vec = vec![vec![]; n];

				// dfs ,返回以root为根节点的树, 节点数量
        fn dfs(root_parent:i32, root:i32, edge_vec : &Vec<Vec<i32>>, child_vec: &mut Vec<Vec<i32>>, tree_count: &mut Vec<i32>) -> i32 {
            let mut count = 1;
          	// 遍历root所有的关联节点
            for &child in &edge_vec[root as usize] {
              	// 过滤掉root的父节点
                if child == root_parent {
                    continue
                }
              	// 其他的就都是root的子节点
                child_vec[root as usize].push(child);
              	// 遍历dfs, 得到所有子节点的树节点数量
                count += dfs(root, child, edge_vec, child_vec, tree_count);
            }
            tree_count[root as usize] = count;
            count
        }
        dfs(0, 0, &edge_vec, &mut child_vec, &mut tree_count);

        let mut res = 0;
        for i in 0..n {
            let child = &child_vec[i];
            if child.len() == 0 || child.len() == 1 {
                res += 1;
            } else {
                let tmp = tree_count[child[0] as usize];
                let mut flag = true;
                for &ch in child {
                    if tree_count[ch as usize] != tmp {
                        flag = false;
                        break
                    }
                }
                if flag {
                    res += 1;
                }
            }
        }
        res
    }
}
```



主要就是通过dfs遍历树, 得到每个子树的节点数量



官解代码简单的多, 也更容易理解, 思路是类似的, 不过在dfs过程中没有想到可以同时做子树节点数量是否相等的校验, 导致代码很不优雅.

```rust
impl Solution {
    pub fn count_good_nodes(edges: Vec<Vec<i32>>) -> i32 {
        let n = edges.len() + 1;
        // 构建领接表
        let mut g = vec![vec![]; n];
        let mut res = 0;
        for edge in edges {
            let (x, y) = (edge[0] as usize, edge[1] as usize);
            g[x].push(y);
            g[y].push(x);
        }

        fn dfs(node: usize, parent: isize, g: &Vec<Vec<usize>>, res: &mut i32) -> usize {
            let mut valid = true;
            let mut tree_size = 0;
            let mut sub_tree_size = 0;

            for &child in &g[node] {
                // 过滤父节点
                if child != parent as usize {
                    // 得到子节点树的数量
                    let size = dfs(child, node as isize, g, res);
                    if sub_tree_size == 0 {
                        sub_tree_size = size;
                    } else if size != sub_tree_size {
                        valid = false;
                    }
                    tree_size += size;
                }
            }
            if valid {
                *res += 1;
            }
            tree_size + 1
        }

        dfs(0, -1, &g, &mut res);
        res
    }
}
```
