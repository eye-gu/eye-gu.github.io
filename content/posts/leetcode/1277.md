## 题目

https://leetcode.cn/problems/count-square-submatrices-with-all-ones/description/

## 自答

```go
func countSquares(matrix [][]int) int {
	m := len(matrix)
	n := len(matrix[0])

	// dp[i][j]代表以i,j点为右下角的最大正方形
	dp := make([][]int, m)
	for i := range dp {
		dp[i] = make([]int, n)
	}

	result := 0
	for i := 0; i < m; i++ {
		for j := 0; j < n; j++ {
			if matrix[i][j] == 0 {
				continue
			}
			if i == 0 || j == 0 || dp[i-1][j-1] == 0 {
				dp[i][j] = 1
				result++
				continue
			}
			dp[i][j] = help(matrix, i, j, dp[i-1][j-1])
			result += dp[i][j]
		}
	}
	return result
}

// 计算i,j为右下角的最大正方形
// k是i-1,j-1的最大正方形, 因此结果必然要小于等于k+1
func help(matrix [][]int, i, j, k int) int {
	result := k + 1
	for n := i - 1; n >= i-k; n-- {
		if matrix[n][j] == 0 {
			result = i - n
			break
		}
	}
	for n := j - 1; n >= j-k; n-- {
		if matrix[i][n] == 0 {
			result = min(result, j-n)
			break
		}
	}
	return result
}
```



## 官解

之前更新dp, 还要遍历i,j点上方和左侧的点是否为1

原来可以直接用`dp[i-1][j]`和``dp[i][j-1]``, 没想到了

```go
func countSquares(matrix [][]int) int {
	m := len(matrix)
	n := len(matrix[0])

	// dp[i][j]代表以i,j点为右下角的最大正方形
	dp := make([][]int, m)
	for i := range dp {
		dp[i] = make([]int, n)
	}

	result := 0
	for i := 0; i < m; i++ {
		for j := 0; j < n; j++ {
			if matrix[i][j] == 0 {
				continue
			}
			if i == 0 || j == 0 || dp[i-1][j-1] == 0 {
				dp[i][j] = 1
				result++
				continue
			}
			// i,j最大正方形
			// 肯定小于i-1,j最大正方形边长
			dp[i][j] = min(dp[i-1][j], dp[i][j-1], dp[i-1][j-1]) + 1
			result += dp[i][j]
		}
	}
	return result
}
```
