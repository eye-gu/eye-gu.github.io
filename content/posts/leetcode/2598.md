https://leetcode.cn/problems/smallest-missing-non-negative-integer-after-operations/description/?envType=daily-question&envId=2025-10-16



```java
class Solution {
        public int findSmallestInteger(int[] nums, int value) {
            // 官解用数组来存, 因为key只有0-value
            Map<Integer, Integer> count = new HashMap<>();
            for (int num : nums) {
                num = num % value;
                // 负数取模结果为负数, 需要加上value
                if (num < 0) {
                    num += value;
                }
                count.put(num, count.getOrDefault(num, 0) + 1);
            }
            for (int i = 0; ; i++) {
                Integer j = count.get(i % value);
                // 找不到i%value的元素,说明不能得到i
                if (j == null) {
                    return i;
                }
                if (j == 1) {
                    count.remove(i % value);
                } else {
                    count.put(i % value, j - 1);
                }
            }
        }
    }
```
