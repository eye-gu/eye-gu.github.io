https://leetcode.cn/problems/find-the-count-of-monotonic-pairs-ii/description/



```rust
impl Solution {
    pub fn count_of_pairs(nums: Vec<i32>) -> i32 {
        let &max = nums.iter().max().unwrap();
        let max = max as usize;
        let n = nums.len();
        let MOD = 1000000007;
        // dp[i][j] 表示arr1[i]的值为j时, 数组的数量
        let mut dp = vec![vec![0; max + 1]; n];

        // 第一个位置上, 只要arr1[0] >= 0 && arr1[0] <= nums[0], 都可以
        for i in 0..=nums[0] as usize {
            dp[0][i] = 1;
        }

        for i in 1..n {
            // 假设nums[i-1]=4, nums[i]=6
            // arr1[i]必须大于等于2, arr2[i]才会小于等于4, 才能保证arr2[i-1] >= arr2[i]
            let d = std::cmp::max(0, nums[i] - nums[i - 1]);
            // 当前位置上可以的取值范围
            for j in d..=nums[i] {
                if j == 0 {
                    // arr1[i]=0, 那么arr1[i-1]只能为0
                    dp[i][0] = dp[i - 1][0];
                } else {
                    // todo!!
                    // 假设nums[i-1]=4, nums[i]=6, j=5
                    // 那么j-d=3, dp[i - 1][j - d] 代表arr1[i-1] = 3, (arr2[i-1] = 1) <= (arr2[i] = 1)
                    dp[i][j as usize] = (dp[i][(j - 1) as usize] + dp[i - 1][(j - d) as usize]) % MOD;
                }
            }
        }

        let mut res = 0;
        for &v in &dp[n - 1] {
            res = (res + v ) % MOD;
        }
        res
    }
}
```
