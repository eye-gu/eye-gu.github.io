```rust
use std::collections::BinaryHeap;

impl Solution {
    pub fn minimum_cost(m: i32, n: i32, horizontal_cut: Vec<i32>, vertical_cut: Vec<i32>) -> i32 {
        let mut heap = BinaryHeap::new();
        for cost in horizontal_cut {
            heap.push((cost, true));
        }
        for cost in vertical_cut {
            heap.push((cost, false));
        }

        let mut row = 1;
        let mut col = 1;
        let mut res = 0;
        // 从大到小遍历
        while let Some((cost, flag)) = heap.pop() {
            // 按行切, 该行成本乘以当前列的数量
            if flag {
                res += cost * col;
                // 行的数量加1
                row += 1;
            // 按列切, 该列成本乘以当前行的数量
            } else {
                res += cost * row;
                // 列的数量加1
                col += 1;
            }
        }

        res
    }
}
```
