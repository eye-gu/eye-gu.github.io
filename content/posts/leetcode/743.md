https://leetcode.cn/problems/network-delay-time/description



```rust
use std::collections::HashMap;

impl Solution {
    pub fn network_delay_time(times: Vec<Vec<i32>>, n: i32, k: i32) -> i32 {
      	// 邻接表
        let mut g = vec![HashMap::new(); n as usize];
        for time in times {
            let map = &mut g[time[0] as usize - 1];
            let x = map.entry(time[1]).or_insert(time[2]);
            if *x > time[2] {
                *x = time[2];
            }
        }
      	// reach[i]表示i节点到达的时间
      	// -1标识没有到达过
        let mut reach = vec![-1; n as usize];
				
        let mut queue = vec![];
        queue.push((k, 0));
        while let Some((x, time)) = queue.pop() {
          	// 过滤掉到达的时间更长
            if reach[x as usize - 1] != -1 && reach[x as usize - 1] <= time {
                continue;
            }
            reach[x as usize - 1] = time;
          	// 广度优先, 把所有可以到达的节点放到队列中
            let map = &g[x as usize - 1];
            for (&k, &v) in map {
                queue.push((k, v + time));
            }
        }

        let mut max = 0;
        for r in reach {
            if r == -1 {
                return -1;
            }
            max = max.max(r);
        }
        max
    }
}
```

官解的Dijkstra算法



想象一张图, 图中有很多节点和有向边, 最开始只有k节点是点亮的, 每次贪心找到当前**所有点亮的点**可以触达的**未点亮点**中最短的那条路径, 然后把该点点亮, 继续这个步骤, 一个个点亮所有点

```java
class Solution {
    public int networkDelayTime(int[][] times, int n, int k) {
        final int INF = Integer.MAX_VALUE / 2;
      	// 邻接表
        int[][] g = new int[n][n];
        for (int i = 0; i < n; ++i) {
            Arrays.fill(g[i], INF);
        }
        for (int[] t : times) {
            int x = t[0] - 1, y = t[1] - 1;
            g[x][y] = t[2];
        }
				// dist[i] 表示从k到i节点的最短耗时
        int[] dist = new int[n];
        Arrays.fill(dist, INF);
      	// 初始节点为0
        dist[k - 1] = 0;
      	// 节点是否被使用过
        boolean[] used = new boolean[n];
        for (int i = 0; i < n; ++i) {
            int x = -1;
          	// 遍历找到没有被使用过的节点中, 从k节点到达最短的节点
          	// 第一次就是k节点本身
          	// 第二次就是k节点可以直接到达的最短的节点, 设其为k2
          	// 第三次就是k加上k2, 这两个点可以触达的点中, 最短的节点
            for (int y = 0; y < n; ++y) {
                if (!used[y] && (x == -1 || dist[y] < dist[x])) {
                    x = y;
                }
            }
            used[x] = true;
          	// 更新所有节点从k节点出发的最短耗时
          	// dist[x] + g[x][y] = 从k到x的最短耗时, 加上x到y的耗时
            for (int y = 0; y < n; ++y) {
                dist[y] = Math.min(dist[y], dist[x] + g[x][y]);
            }
        }

        int ans = Arrays.stream(dist).max().getAsInt();
        return ans == INF ? -1 : ans;
    }
}
```
