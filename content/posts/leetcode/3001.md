https://leetcode.cn/problems/minimum-moves-to-capture-the-queen/description/



```rust
impl Solution {
    pub fn min_moves_to_capture_the_queen(a: i32, b: i32, c: i32, d: i32, e: i32, f: i32) -> i32 {
        // 车可以直接攻击皇后
        if a == e {
            // 象在中间不行
            if c == a {
                if d > b && d < f {
                    return 2;
                }
                if d > f && d < b {
                    return 2;
                }
            }
            return 1;
        }
        if b == f {
            if d == f {
                if c > a && c < e {
                    return 2;
                }
                if c > e && c < a {
                    return 2;
                }
            }

            return 1;
        }
        // 象和皇后在同一斜线
        if (c - e).abs() == (d - f).abs() {
            // 车和皇后在同一斜线, 并且在斜线的中间
            if (a - e).abs() == (b - f).abs() { 
                if a > c && a < e && b > d && b < f {
                    return 2;
                }
                if a > c && a < e && b < d && b > f {
                    return 2;
                }
                if a < c && a > e && b < d && b > f {
                    return 2;
                }
                if a < c && a > e && b > d && b < f {
                    return 2;
                }
            }
            return 1;
        }
        2
    }
}
```

官解:

```rust
use std::cmp::{min, max};

impl Solution {
    pub fn min_moves_to_capture_the_queen(a: i32, b: i32, c: i32, d: i32, e: i32, f: i32) -> i32 {
        // 车与皇后处在同一行，且中间没有象
        if a == e && (c != a || d <= min(b, f) || d >= max(b, f)) {
            return 1;
        }
        // 车与皇后处在同一列，且中间没有象
        if b == f && (d != b || c <= min(a, e) || c >= max(a, e)) {
            return 1;
        }
        // 象、皇后处在同一条对角线，且中间没有车
        if (c - e).abs() == (d - f).abs() && ((c - e) * (b - f) != (a - e) * (d - f)
            || a < min(c, e) || a > max(c, e)) {
            return 1;
        }
        2
    }
}
```



优雅的写法:

1. 表示a不在b,c中间可以用`a <= min(b, c) || a >= max(b, c)`

2. `(c - e) * (b - f) != (a - e) * (d - f)`转换一下`(a - e)/(b - f) != (c - e)/(d - f)`, 就是不在同一斜线上
