[3201. 找出有效子序列的最大长度 I - 力扣（LeetCode）](https://leetcode.cn/problems/find-the-maximum-length-of-valid-subsequence-i/description/)

```java
class Solution {
        public int maximumLength(int[] nums) {
            // 只有三种可能, 全部都是奇数, 全部是偶数,奇偶间隔
            int zeroCount = 0;
            int oneCount = 0;
            int intervalCount = 1;
            boolean zeroFlag = nums[0] % 2 == 0;
            for (int i = 0; i < nums.length; i++) {
                if (nums[i] % 2 == 0) {
                    zeroCount++;
                    if (!zeroFlag) {
                        zeroFlag = true;
                        intervalCount++;
                    }
                } else {
                    oneCount++;
                    if (zeroFlag) {
                        zeroFlag = false;
                        intervalCount++;
                    }
                }
            }
            return Math.max(intervalCount, Math.max(zeroCount, oneCount));
        }
    }
```
