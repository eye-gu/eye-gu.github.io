impl Solution {
  pub fn min_cut(s: String) -> i32 {
    let n = s.len();
    // dp[i][j]代表[i, j]是否是回文串
    let mut dp = vec![vec![false; n]; n];
    for i in 0..n {
      dp[i][i] = true
    }
    let chars:Vec<_> = s.chars().collect();
    // 这里逆序遍历, 因为dp[i][j]依赖于dp[i+1][j-1];
    for i in (0..n).rev() {
      for j in i+1..n {
        if chars[i] == chars[j] {
          // 只有两个字符
          if i + 1 == j {
            dp[i][j] = true;
          } else {
            // 内部是回文串
            dp[i][j] = dp[i+1][j-1];
          }
        }
      }
    }

    // dp_min_cut[i]代表[0, i]的最小分割次数
    let mut dp_min_cut = vec![n as i32; n];
    for i in 0..n {
      if dp[0][i] {
        // 整体是回文串, 不需要分割
        dp_min_cut[i] = 0;
      } else {
        // 遍历所有以i为结尾的回文串, 找到最小的分割次数
        for j in 1..=i {
          if dp[j][i] {
            dp_min_cut[i] = dp_min_cut[i].min(dp_min_cut[j-1] + 1)
          }
        }
      }
    }

    dp_min_cut[n - 1]
  }
}
