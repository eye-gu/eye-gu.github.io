impl Solution {
    pub fn shortest_distance_after_queries(n: i32, queries: Vec<Vec<i32>>) -> Vec<i32> {
        let n = n as usize;
      	// dp[i] 表示i到n-1的最短距离
        let mut dp = vec![n as i32; n];
        dp[n - 1] = 0;
      	// g是领接表
        let mut g = vec![vec![]; n];

        fn shortest_distance(dp: &mut Vec<i32>, i: usize, g:&Vec<Vec<usize>>) {
          	// 从i往前遍历, 更新其到n-1的最短距离
            for j in (0..=i).rev() {
                // 每个节点的最短距离, 要不就是直接到下一个节点+1
                dp[j] = dp[j].min(dp[j+1] + 1);
              	// 要不就是所有有通路的下个节点+1
                for &k in &g[j] {
                    dp[j] = dp[j].min(dp[k] + 1);
                }
            }
        }
				// 初始, n-2到n-1为1, n-3到n-1为2...
        shortest_distance(&mut dp, n - 2, &g);
        let mut res = Vec::with_capacity(queries.len());
        for query in queries {
            g[query[0] as usize].push(query[1] as usize);
          	// 每修一条路, 只影响路的开始节点query[0]及其前面的节点到n-1的路程
            shortest_distance(&mut dp, query[0] as usize, &g);
            res.push(dp[0]);
        }

        res
    }
}
