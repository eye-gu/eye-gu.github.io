[2322. 从树中删除边的最小分数 - 力扣（LeetCode）](https://leetcode.cn/problems/minimum-score-after-removals-on-a-tree/description/)



官解.

1. clock, in, out来判断两个节点是否是父子树关系
2. 异或的特殊性, a是由b和c异或得到, 那么在得到a和c的情况下, b可以通过a异或c得到. 即`a = b xor c`, `b = a xor c`

```go
func minimumScore(nums []int, edges [][]int) int {
	n := len(nums)
	g := make([][]int, n)
	for _, e := range edges {
		g[e[0]] = append(g[e[0]], e[1])
		g[e[1]] = append(g[e[1]], e[0])
	}
	in := make([]int, n)
	out := make([]int, n)
	xor := make([]int, n)
	clock := 0
	dfs(0, -1, g, in, out, xor, nums, &clock)

	ans := 1 << 30
	for i := 1; i < n; i++ {
		for j := i + 1; j < n; j++ {
			if in[i] < in[j] && in[j] <= out[i] {
				// j是i的子树
				a := xor[i] ^ xor[j]
				b := xor[j]
				c := xor[0] ^ xor[i]
				ans = min(ans, max(a, max(b, c))-min(a, min(b, c)))
			} else if in[j] < in[i] && in[i] <= out[j] {
				// i是j的子树
				a := xor[j] ^ xor[i]
				b := xor[i]
				c := xor[0] ^ xor[j]
				ans = min(ans, max(a, max(b, c))-min(a, min(b, c)))
			} else {
				// i,j是两颗树
				a := xor[i]
				b := xor[j]
				c := xor[0] ^ xor[j] ^ xor[i]
				ans = min(ans, max(a, max(b, c))-min(a, min(b, c)))
			}
		}
	}
	return ans
}

func dfs(u int, fa int, g [][]int, in, out, xor, nums []int, clock *int) {
	*clock++
	in[u] = *clock
	xor[u] = nums[u]
	for _, v := range g[u] {
		if v == fa {
			continue
		}
		dfs(v, u, g, in, out, xor, nums, clock)
		xor[u] ^= xor[v]
	}
	out[u] = *clock
}
```
