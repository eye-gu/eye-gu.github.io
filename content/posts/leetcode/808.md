https://leetcode.cn/problems/soup-servings/description/



一开始还以为是数学题, 看了官解才知道可以用dp.

这个思路, 还有这个初始状态, 还有这个状态转移, 好难想啊.

```rust
impl Solution {
    pub fn soup_servings(n: i32) -> f64 {
        let n = (n as usize + 24) / 25;
        if n >= 179 {
            return 1.0;
        }
        let mut dp = vec![vec![0f64; n + 1]; n + 1];
        dp[0][0] = 0.5;
        for i in 1..=n {
            dp[0][i] = 1.0;
        }

        for i in 1..=n {
            for j in 1..=n {
                dp[i][j] = 0.25 * (dp[i.saturating_sub(1)][j.saturating_sub(3)]
                    + dp[i.saturating_sub(2)][j.saturating_sub(2)]
                    + dp[i.saturating_sub(3)][j.saturating_sub(1)]
                    + dp[i.saturating_sub(4)][j]
                );
            }
        }

        dp[n][n]
    }
}
```
