```rust
use std::collections::{BTreeSet, HashMap};
use std::cmp::Ordering;
use std::hash::Hash;

struct ExamRoom {
    n: i32,
    segments: BTreeSet<Segment>,
    // 线段的左端点 -> 线段的右端点
    start_end: HashMap<i32, i32>,
    // 线段的右端点 -> 线段的左端点
    end_start: HashMap<i32, i32>,
}

struct Segment {
    // 座位线段的左端点, 不包含
    start: i32,
    // 座位线段的右端点, 不包含
    end: i32,
    // 线段中间的座位
    seat: i32,
    // 线段中间的座位离左端点的间隔
    max_interval: i32,
}

impl Segment {
    fn new(start: i32, end: i32, n:i32) -> Segment {
        let seat;
        let max_interval;
        // 左端点-1, 就选0号座位
        if start == -1 {
            seat = 0;
            max_interval = end - start - 1;
        // 右端点n, 就选n-1座位
        } else if end == n  {
            seat = n - 1;
            max_interval = end - start - 1;
        // 否则选中间
        } else {
            seat = start + (end - start) / 2;
            max_interval = (end - start) / 2;
        }
        Segment {
            start,
            end,
            seat,
            max_interval,
        }
    }
}

impl Ord for Segment {
    fn cmp(&self, other: &Self) -> Ordering {
        // 根据间隔排序, 间隔一致根据座位的大小排序
        if other.max_interval == self.max_interval {
            return self.seat.cmp(&other.seat);
        }
        other.max_interval.cmp(&self.max_interval)
    }
}
impl PartialOrd for Segment {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
    
}
impl Eq for Segment {
    
}
impl PartialEq for Segment {
    fn eq(&self, other: &Self) -> bool {
        self.start == other.start && self.end == other.end
    }

}

impl ExamRoom {

    fn new(n: i32) -> Self {
        let mut segments = BTreeSet::new();
        segments.insert(Segment::new(-1, n, n));
        let mut start_end = HashMap::new();
        let mut end_start = HashMap::new();
        start_end.insert(-1, n);
        end_start.insert(n, -1);
        ExamRoom { n, segments , start_end, end_start }
    }
    
    fn seat(&mut self) -> i32 {
        let segment = self.segments.pop_first().unwrap();
        self.start_end.remove(&segment.start);
        self.end_start.remove(&segment.end);
        let seat = segment.seat;
        self.segments.insert(Segment::new(segment.start, seat, self.n));
        self.start_end.insert(segment.start, seat);
        self.end_start.insert(seat, segment.start);
        self.segments.insert(Segment::new(seat, segment.end, self.n));
        self.start_end.insert(seat, segment.end);
        self.end_start.insert(segment.end, seat);
        seat
    }
    
    fn leave(&mut self, p: i32) {
        // 找到p点的左右两个线段
        let right = self.start_end.remove(&p).unwrap();
        let left = self.end_start.remove(&p).unwrap();
        self.segments.remove(&Segment::new(left, p, self.n));
        self.segments.remove(&Segment::new(p, right, self.n));
        self.end_start.remove(&right);
        self.start_end.remove(&left);
        self.segments.insert(Segment::new(left, right, self.n));
        self.start_end.insert(left, right);
        self.end_start.insert(right, left);
    }
}
```

使用一个seats存放已经被占用的座位, 来替代start_end和end_start. 因为seats是BTreeSet类型, 可以方便的查到某个座位号左边被占用的座位和右边被占用的座位.

```rust
use std::cmp::Ordering;
use std::hash::Hash;

struct ExamRoom {
    n: i32,
    segments: BTreeSet<Segment>,
    // 已经被占用的座位
    seats: BTreeSet<i32>,
}

struct Segment {
    // 座位线段的左端点, 不包含
    start: i32,
    // 座位线段的右端点, 不包含

    seat: i32,

        let max_interval;

            seat = 0;
            max_interval = end - start - 1;

            start,
            end,

    }
    
}

}

    fn eq(&self, other: &Self) -> bool {
        self.start == other.start && self.end == other.end
    }

/** 
 * `&self` means the method takes an immutable reference.
 * If you need a mutable reference, change it to `&mut self` instead.
 */
impl ExamRoom {

    fn new(n: i32) -> Self {
        let mut segments = BTreeSet::new();
        segments.insert(Segment::new(-1, n, n));
        let mut seats = BTreeSet::new();
        seats.insert(-1);
        seats.insert(n);
        ExamRoom { n, segments , seats }
    }
    
    fn seat(&mut self) -> i32 {
        let segment = self.segments.pop_first().unwrap();
        let seat = segment.seat;
        self.segments.insert(Segment::new(segment.start, seat, self.n));
        self.segments.insert(Segment::new(seat, segment.end, self.n));
        self.seats.insert(seat);
        seat
    }
    
    fn leave(&mut self, p: i32) {
        // 找到p点的左右两个线段
        let &left = self.seats.range(..p).next_back().unwrap();
        let &right = self.seats.range(p+1..).next().unwrap();
        self.segments.remove(&Segment::new(left, p, self.n));
        self.segments.remove(&Segment::new(p, right, self.n));
        self.seats.remove(&p);
        self.segments.insert(Segment::new(left, right, self.n));
    }
}
```
