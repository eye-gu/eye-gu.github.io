```rust
impl Solution {
    pub fn total_n_queens(n: i32) -> i32 {
        fn dfs(rows: &mut Vec<usize>, row: usize, n: usize) -> i32 {
            if row == n {
                return 1;
            }
            let mut res = 0;
            // 遍历所有位置, 判断是否能放
            for i in 0..n {
                if check(rows, row, i) {
                    rows[row] = i;
                    res += dfs(rows, row + 1, n);
                }
            }

            res
        }

        fn check(rows: &Vec<usize>, row: usize, col: usize) -> bool {
            // 遍历前几行, 列是否相等, 列差和行差是否相等
            for j in 0..row {
                let k = rows[j];
                if col > k {
                    if row - j == col - k {
                        return false;
                    }
                } else if col < k {
                    if row - j == k - col {
                        return false;
                    }
                } else if col == k {
                    return false;
                }
                
            }
            return true;
        }

        let mut rows = vec![0; n as usize];
        dfs(&mut rows, 0, n as usize)
    }
}
```
