https://leetcode.cn/problems/find-the-original-typed-string-ii

超出了内存限制
```java
class Solution {
        public int possibleStringCount(String word, int k) {
            int mod = 1000000007;
            int n = word.length();
            // 每个相同字符组的数量
            List<Integer> counts = new ArrayList<>();
            int count = 1;
            for (int i = 1; i < n; i++) {
                if (word.charAt(i) == word.charAt(i - 1)) {
                    count++;
                } else {
                    counts.add(count);
                    count = 1;
                }
            }
            counts.add(count);
            //dp[i][j]代表第i个相同字符组，长度为j的数量
            int[][] dp = new int[counts.size()][n + 1];
            count = counts.get(0);
            for (int i = 1; i <= count; i++) {
                dp[0][i] = 1;
            }
            for (int i = 1; i < counts.size(); i++) {
                count = counts.get(i);
                for (int j = 1; j <= count; j++) {
                    for (int m = j + 1; m <= n; m++) {
                        // i组提供j个字符，那么长度为m时，i-1组的长度为m-j
                        dp[i][m] = (dp[i][m] + dp[i - 1][m - j]) % mod;
                    }
                }
            }
            int ans = 0;
            for (int i=k; i<=n; i++) {
                ans = (ans + dp[counts.size() - 1][i]) % mod;
            }
            return ans;
        }
    }
```
