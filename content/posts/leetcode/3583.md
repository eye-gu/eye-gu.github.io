https://leetcode.cn/problems/count-special-triplets/description/?envType=daily-question&envId=2025-12-09



```java
class Solution {
        public int specialTriplets(int[] nums) {
            // 前面的数字出现的次数
            Map<Integer, Integer> preCount = new HashMap<>();
            // 前面的数字出现的次数 * 2 是特殊的三元组的和
            Map<Integer, Integer> preSpecialTripletsSum = new HashMap<>();
            int mod = 1_000_000_007;
            int ans = 0;
            for (int num : nums) {
                // 当前数字作为第三个数字的情况
                ans = (ans + preSpecialTripletsSum.getOrDefault(num, 0)) % mod;
                // 当前数字作为第二个数字的情况
                int doubleNum = num * 2;
                int doubleNumCount = preCount.getOrDefault(doubleNum, 0);
                preSpecialTripletsSum.put(doubleNum, (doubleNumCount + preSpecialTripletsSum.getOrDefault(doubleNum, 0)) % mod);

                preCount.put(num, preCount.getOrDefault(num, 0) + 1);
            }
            return ans;
        }
    }
```
