https://leetcode.cn/problems/maximum-frequency-of-an-element-after-performing-operations-ii/description/?envType=daily-question&envId=2025-10-22



基准值只考虑了nums中的数字, 没有考虑到可以两个数字的中间值, 例如[5, 62],k=40, 两次操作变成[30,30]的情况

```java
class Solution {

    public int maxFrequency(int[] nums, int k, int numOperations) {
        Arrays.sort(nums);
        int n = nums.length;
        int maxFreq = 0;

        // 统计每个数字的出现频率
        Map<Integer, Integer> freqMap = new HashMap<>();
        for (int num : nums) {
            freqMap.put(num, freqMap.getOrDefault(num, 0) + 1);
        }

        int left = 0;
        int right = 0;
        // 滑动窗口，考虑每个可能的基准值
        for (int i = 0; i < n; i++) {
            // 找到所有可以通过操作变为nums[i]的元素
            while (left < i && nums[i] - nums[left] >= k) {
                left++;
            }

            while (right < n && nums[right] - nums[i] <= k) {
                right++;
            }

            // 窗口内元素总数
            int windowSize = right - left;

            // 已经等于nums[i]的元素数量
            int sameCount = freqMap.get(nums[i]);

            // 需要操作的元素数量
            int needOps = windowSize - sameCount;

            // 实际可以操作的元素数量
            int actualOps = Math.min(needOps, numOperations);

            // 当前频率 = 已存在数量 + 实际操作数量
            int currentFreq = sameCount + actualOps;

            maxFreq = Math.max(maxFreq, currentFreq);
        }

        return maxFreq;
    }
}
```



官解. 对区间值同样进行了考虑

```java
class Solution {
        public int maxFrequency(int[] nums, int k, int numOperations) {
            Arrays.sort(nums);
            int ans = 0;
            // 统计每个数字出现的次数
            Map<Integer, Integer> numCount = new HashMap<>();
            Set<Integer> modes = new TreeSet<>();

            // 原始值, 减去k, 加上k, 都有可能作为基准值
            Consumer<Integer> addMode = (value) -> {
                modes.add(value);
                if (value - k >= nums[0]) {
                    modes.add(value - k);
                }
                if (value + k <= nums[nums.length - 1]) {
                    modes.add(value + k);
                }
            };
            
            int lastNumIndex = 0;
            for (int i = 0; i < nums.length; ++i) {
                // 数字不同了, 
                if (nums[i] != nums[lastNumIndex]) {
                    numCount.put(nums[lastNumIndex], i - lastNumIndex);
                    ans = Math.max(ans, i - lastNumIndex);
                    addMode.accept(nums[lastNumIndex]);
                    lastNumIndex = i;
                }
            }
            
            numCount.put(nums[lastNumIndex], nums.length - lastNumIndex);
            ans = Math.max(ans, nums.length - lastNumIndex);
            addMode.accept(nums[lastNumIndex]);
            
            // 二分查找
            IntUnaryOperator leftBound = (value) -> {
                int left = 0, right = nums.length - 1;
                while (left < right) {
                    int mid = (left + right) / 2;
                    if (nums[mid] < value) {
                        left = mid + 1;
                    } else {
                        right = mid;
                    }
                }
                return left;
            };
            // 二分查找
            IntUnaryOperator rightBound = (value) -> {
                int left = 0, right = nums.length - 1;
                while (left < right) {
                    int mid = (left + right + 1) / 2;
                    if (nums[mid] > value) {
                        right = mid - 1;
                    } else {
                        left = mid;
                    }
                }
                return left;
            };
            
            for (int mode : modes) {
                // 找到以mode为基准值的左右边界
                int l = leftBound.applyAsInt(mode - k);
                int r = rightBound.applyAsInt(mode + k);
                int tempAns;
                if (numCount.containsKey(mode)) {
                    tempAns = Math.min(r - l + 1, numCount.get(mode) + numOperations);
                } else {
                    tempAns = Math.min(r - l + 1, numOperations);
                }
                ans = Math.max(ans, tempAns);
            }
            
            return ans;
        }
    }
```
