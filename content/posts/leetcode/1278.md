https://leetcode.cn/problems/palindrome-partitioning-iii/description/



```rust
use std::cmp::min;
impl Solution {
  pub fn palindrome_partition(s: String, k: i32) -> i32 {
    // dp[i][j]代表s[0..i]分割成j个回文子串, 需修改的最少字符数
    let mut dp = vec![vec![s.len() as i32; (k + 1) as usize]; s.len()];

    // 将s[i..j]修改成回文串需要的操作数
    fn palindrome(s: &[char], mut i: usize, mut  j: usize) -> i32 {
      let mut count = 0;
      while i <= j && j > 0 {
        if s[i] != s[j] {
          count += 1;
        }
        i += 1;
        j -= 1;
      }
      count
    }

    let s = s.chars().collect::<Vec<_>>();
    for i in 0..s.len() {
      // 改成一个回文串的操作数
      dp[i][1] = palindrome(&s, 0, i);
      for j in 2..=min(k as usize, i + 1) {
        // dp[i][j]的最少操作数, 等于min(s[0..k]改成j-1的最少操作数, 加上s[k+1,i]改成回文串的操作数)
        for k in 0..i {
          dp[i][j] = dp[i][j].min(dp[k][j-1] + palindrome(&s, k+1, i))
        }
      }
    }
    dp[s.len() -1][k as usize]
  }
}
```

根据官解, 还可以优化, 提前计算s[i..j]修改成回文串的修改次数

```rust
use std::cmp::min;
impl Solution {
  pub fn palindrome_partition(s: String, k: i32) -> i32 {
    // dp[i][j]代表s[0..i]分割成j个回文子串, 需修改的最少字符数
    let mut dp = vec![vec![s.len() as i32; (k + 1) as usize]; s.len()];

    let s = s.chars().collect::<Vec<_>>();

    // cost[i][j]代表s[i..j]修改成回文串的修改次数
    let mut cost = vec![vec![0; s.len()]; s.len()];
    for i in (0..s.len()).rev() {
      for j in i..s.len() {
        if i == j {
          cost[i][j] = 0;
        } else if j - i == 1 {
          cost[i][j] = if s[i] == s[j] { 0 } else { 1 };
        } else {
          cost[i][j] = cost[i+1][j-1] + if s[i] == s[j] { 0 } else { 1 }
        }
      }
    }

    for i in 0..s.len() {
      // 改成一个回文串的操作数
      dp[i][1] = cost[0][i];
      for j in 2..=min(k as usize, i + 1) {
        // dp[i][j]的最少操作数, 等于min(s[0..k]改成j-1的最少操作数, 加上s[k+1,i]改成回文串的操作数)
        for k in 0..i {
          dp[i][j] = dp[i][j].min(dp[k][j-1] + cost[k+1][i])
        }
      }
    }
    dp[s.len() -1][k as usize]
  }
}
```
