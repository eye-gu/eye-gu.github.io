https://leetcode.cn/problems/design-a-number-container-system/submissions/663476262/



一开始的思路很简单, 就是维护索引->数字, 数字->索引列表这两个映射, 但是超时了

```java
class NumberContainers {

    private Map<Integer, Integer> indexToNumber = new HashMap<>();
    private Map<Integer, PriorityQueue<Integer>> numberToIndex = new HashMap<>();

    public NumberContainers() {

    }

    public void change(int index, int number) {
        Integer old = indexToNumber.get(index);
        if (old != null) {
            PriorityQueue<Integer> oldIndexes = numberToIndex.get(old);
            oldIndexes.remove(index);
        }
      	indexToNumber.put(index, number);
        PriorityQueue<Integer> indexes = numberToIndex.computeIfAbsent(number, k -> new PriorityQueue<>());
        indexes.add(index);
    }

    public int find(int number) {
        PriorityQueue<Integer> indexes = numberToIndex.get(number);
        if (indexes == null || indexes.isEmpty()) {
            return -1;
        }
        return indexes.peek();
    }
}
```



官解的延时删除, 有点意思

```java
class NumberContainers {
    private Map<Integer, Integer> nums = new HashMap<>();
    private Map<Integer, PriorityQueue<Integer>> heaps = new HashMap<>();

    public void change(int index, int number) {
        nums.put(index, number);
        heaps.computeIfAbsent(number, k -> new PriorityQueue<>()).add(index);
    }

    public int find(int number) {
        PriorityQueue<Integer> heap = heaps.get(number);
        if (heap == null) {
            return -1;
        }
        while (!heap.isEmpty() && !nums.get(heap.peek()).equals(number)) {
            heap.poll();
        }
        return heap.isEmpty() ? -1 : heap.peek();
    }
}
```
