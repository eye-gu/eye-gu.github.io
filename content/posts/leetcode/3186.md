https://leetcode.cn/problems/maximum-total-damage-with-spell-casting/description/?envType=daily-question&envId=2025-10-11



```java
class Solution {
        public long maximumTotalDamage(int[] power) {
            Map<Integer, Integer> count = new HashMap<>();
            for (int p : power) {
                count.put(p, count.getOrDefault(p, 0) + 1);
            }
            Arrays.sort(power);
            // dp[i]: 在power[i]为止的最大伤害
            long[] dp = new long[power.length];
            dp[0] = ((long) power[0]) * count.get(power[0]);
            int j = 0;
            for (int i = 1; i < power.length; i++) {
                // 相同, 最大伤害也相同
                if (power[i] == power[i-1]) {
                    dp[i] = dp[i-1];
                    continue;
                }
                while (j < i && power[i] - power[j] > 2) {
                    j++;
                }
                // 取左侧的最大伤害, 或者当前为止的伤害加上j-1(j-1位置比当前为止小于2)位置上的最大伤害
                if (j == 0) {
                    dp[i] = Math.max(dp[i-1], (long) power[i] * count.get(power[i]));
                } else {
                    dp[i] = Math.max(dp[i-1], dp[j-1] + (long) power[i] * count.get(power[i]));
                }
            }
            return dp[power.length - 1];
        }
    }
```
