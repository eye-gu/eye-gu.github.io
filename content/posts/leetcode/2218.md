https://leetcode.cn/problems/maximum-value-of-k-coins-from-piles/description/



```python
from typing import List


class Solution:
    def maxValueOfCoins(self, piles: List[List[int]], k: int) -> int:
        # dp[i]表示取i个硬币的最大价值
        dp = [-1] * (k + 1)
        dp[0] = 0
        # 遍历所有硬币栈

        # 从原有方案取k-1, 当前硬币栈取1, 组成k个硬币
        # 从原有方案取k-2, 当前硬币栈取2, 组成k个硬币
        # ...
        # 从原有方案取0, 当前硬币栈取k, 组成k个硬币
        # 上面是所有取k个硬币的所有可能, 取最大值

        # 从原有方案取k-2, 当前硬币栈取1, 组成k-1个硬币
        # 从原有方案取k-3, 当前硬币栈取2, 组成k-1个硬币
        # ...
        # 从原有方案取0, 当前硬币栈取k-1, 组成k-1个硬币
        # 上面是所有取k-1个硬币的所有可能, 取最大值

        #...

        # 从原有方案取1, 当前硬币栈取1, 组成2个硬币
        # 从原有方案取0, 当前硬币栈取2, 组成2个硬币

        # 从原有方案取0, 当前硬币栈取1, 组成1个硬币
        for pile in piles:
            # 从k遍历到1, 代表找所有可以组合成i个硬币的方案
            # 如果从1开始遍历, 那么dp[4]先修改, dp[5]后修改, 但dp[5]修改的时候会使用dp[4]的值
            for i in range(k, 0, -1):
                value = 0
                # 遍历当前硬币栈的每个硬币
                for j in range(1, len(pile)+1):
                    value += pile[j-1]
                    if i < j:
                        break
                    # != -1 代表原有i-j个硬币的方案存在
                    if (dp[i - j] != -1):
                        dp[i] = max(dp[i], dp[i-j] + value)
        return dp[k]


if __name__ == '__main__':
    solution = Solution()
    ans = solution.maxValueOfCoins([[1,100,3],[7,8,9]], 2)
    print(ans)
```
