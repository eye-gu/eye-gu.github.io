```rust
use std::collections::HashMap;

impl Solution {
    pub fn knight_dialer(n: i32) -> i32 {
        const MOD: i32 = 1000000007;
        let next: Vec<Vec<usize>> = vec![
            vec![4, 6],
            vec![6, 8],
            vec![7, 9],
            vec![4, 8],
            vec![3, 9, 0],
            vec![],
            vec![1, 7, 0],
            vec![2, 6],
            vec![1, 3],
            vec![2, 4]
        ];


        fn dfs(cur: usize, n:i32, memo: &mut Vec<HashMap<i32, i32>>, next: &Vec<Vec<usize>>) -> i32 {
            if n == 0 {
                return 1;
            }

            if let Some(&k) = memo[cur].get(&n) {
                return k;
            }

            let mut res = 0;
            for &i in &next[cur] {
                res += dfs(i, n - 1, memo, next);
                res %= MOD;
            }
            memo[cur].insert(n, res);
            res
        }

      	// 下标为当前的号码, key为还有多少个号码
        let mut memo: Vec<HashMap<i32, i32>> = vec![HashMap::new(); 10];

        let mut res = 0;
        for i in 0..=9 {
            res += dfs(i, n -1, &mut memo, &next);
            res %= MOD;
        }

        res
    }
}
```

```rust
const MOD: i32 = 1_000_000_007;

impl Solution {
    pub fn knight_dialer(n: i32) -> i32 {
        let moves = vec![
            vec![4, 6],
            vec![6, 8],
            vec![7, 9],
            vec![4, 8],
            vec![3, 9, 0],
            vec![],
            vec![1, 7, 0],
            vec![2, 6],
            vec![1, 3],
            vec![2, 4],
        ];
        let mut d = vec![vec![0; 10], vec![1; 10]];
        for i in 2..=n {
            let x = (i % 2) as usize;
            for j in 0..10 {
              	// 初始化
                d[x][j] = 0;
              	// 遍历可以走到j的节点
                for &k in &moves[j] {
                  	// 1 - x为上一层的下标
                  	// 上一层的相加
                    d[x][j] = (d[x][j] + d[1 - x][k]) % MOD;
                }
            }
        }
      	// 对应层的累加
        d[(n % 2) as usize].iter().fold(0, |res, &x| (res + x) % MOD)
    }
}
```
