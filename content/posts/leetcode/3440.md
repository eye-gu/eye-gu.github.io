```java
class Solution {
        public int maxFreeTime(int eventTime, int[] startTime, int[] endTime) {
            int n = startTime.length;
            // 空隙长度 -> 空隙索引列表
            TreeMap<Integer, List<Integer>> gapIndexMap = new TreeMap<>();
            gapIndexMap.put(startTime[0], new ArrayList<>());
            gapIndexMap.get(startTime[0]).add(0);
            for (int i = 1; i < n; i++) {
                int gap = startTime[i] - endTime[i - 1];
                List<Integer> indexes = gapIndexMap.computeIfAbsent(gap, k -> new ArrayList<>());
                indexes.add(i);
            }
            int gap = eventTime - endTime[n - 1];
            List<Integer> indexes = gapIndexMap.computeIfAbsent(gap, k -> new ArrayList<>());
            indexes.add(n);

            int ans = 0;
            for (int i = 0; i < n; i++) {
                int pre = i == 0 ? startTime[i] : startTime[i] - endTime[i - 1];
                int post = i == n - 1 ? eventTime - endTime[i] : startTime[i + 1] - endTime[i];
                int len = endTime[i] - startTime[i];
                if (!checkMove(i, len, gapIndexMap)) {
                    ans = Math.max(ans, pre + post);
                } else {
                    ans = Math.max(ans, pre + len + post);
                }
            }
            return ans;
        }

        private boolean checkMove(int i, int len, TreeMap<Integer, List<Integer>> gapIndexMap) {
            NavigableMap<Integer, List<Integer>> tailMap = gapIndexMap.tailMap(len, true);
            for (Map.Entry<Integer, List<Integer>> entry : tailMap.entrySet()) {
                for (int index : entry.getValue()) {
                    // 空隙不在当前会议的左右两边
                    if (i != index && i != index - 1) {
                        return true;
                    }
                }
            }
            return false;
        }
    }
```

依照官解的思路进行优化, 官解一次循环同时遍历左右两边怕以后看不懂, 还是分两遍循环

```java
class Solution {
        public int maxFreeTime(int eventTime, int[] startTime, int[] endTime) {
            int n = startTime.length;
            boolean[] flag = new boolean[n];
            // max记录当前会议前最大的空隙
            int max = 0;
            for (int i = 0; i < n; i++) {
                if (endTime[i] - startTime[i] <= max) {
                    flag[i] = true;
                }
                max = Math.max(max, i == 0 ? startTime[i] : startTime[i] - endTime[i - 1]);
            }
            // max记录当前会议后最大的空隙
            max = 0;
            for (int i=n-1; i>=0; i--) {
                if (endTime[i] - startTime[i] <= max) {
                    flag[i] = true;
                }
                max = Math.max(max, i == n - 1 ? eventTime - endTime[i] : startTime[i + 1] - endTime[i]);
            }

            int ans = 0;
            for (int i = 0; i < n; i++) {
                int pre = i == 0 ? startTime[i] : startTime[i] - endTime[i - 1];
                int post = i == n - 1 ? eventTime - endTime[i] : startTime[i + 1] - endTime[i];
                int len = endTime[i] - startTime[i];
                if (!flag[i]) {
                    ans = Math.max(ans, pre + post);
                } else {
                    ans = Math.max(ans, pre + len + post);
                }
            }
            return ans;
        }
    }
```
