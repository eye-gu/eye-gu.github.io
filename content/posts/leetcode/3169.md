https://leetcode.cn/problems/count-days-without-meetings/description/

```java
class Solution {
        public int countDays(int days, int[][] meetings) {
            // 根据会议开始时间排序
            Arrays.sort(meetings, (a, b) -> a[0] - b[0]);
            // 会议的总时间
            int sum = 0;
            // 当前最晚会议的结束时间
            int end = 0;
            for (int[] meeting : meetings) {
                // 如果会议和当前最晚结束时间重叠, 要么完全重叠, 要么还有部分可以继续开
                if (meeting[0] <= end) {
                    if (meeting[1] > end) {
                        sum += meeting[1] - end;
                        end = meeting[1];
                    }
                } else {
                  // 完全不重叠, 那么可以完整开这个会
                    sum += meeting[1] - meeting[0] + 1;
                    end = meeting[1];
                }
            }
            return days - sum;
        }
    }
```
