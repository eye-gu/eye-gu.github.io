https://leetcode.cn/problems/find-the-minimum-amount-of-time-to-brew-potions/description/?envType=daily-question&envId=2025-10-09



```java
class Solution {

    public long minTime(int[] skill, int[] mana) {
        int n = skill.length;
        int m = mana.length;
        // endTime[i] 表示第i个巫师结束的时间
        long[] endTime = new long[n];
        // 初始化第一瓶魔药的加工结束时间
        endTime[0] = (long) skill[0] * mana[0];
        for (int i = 1; i < n; i++) {
            endTime[i] = endTime[i-1] + (long) skill[i] * mana[0];
        }
        for (int i = 1; i < m; i++) {
            // end记录最后的结束时间
            long end = endTime[0] + (long) skill[0] * mana[i];
            for (int j=1; j<n; j++) {
                // 要么从上个巫师的结束后开始加工, 要么从上个魔药的结束时间开始加工
                end = Math.max(end, endTime[j]) + (long) skill[j] * mana[i];
            }
            endTime[n-1] = end;
            // 再反过来更新每个巫师真正的结束时间
            for (int j=n-2; j>=0; j--) {
                endTime[j] = endTime[j+1] - (long) skill[j + 1] * mana[i];
            }
        }
        return endTime[n-1];
    }
}
```
