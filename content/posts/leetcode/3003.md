```java
public class Solution {

        public int maxPartitionsAfterOperations(String s, int k) {
            int n = s.length();
            // left[i] 存储i的左侧(不包含i)最近的分区的信息：[分区数, 字符掩码, 当前分区不同字符数]
            int[][] left = new int[n][3];
            // right[i] 存储i的右侧(不包含i)最近的分区的信息：[分区数, 字符掩码, 当前分区不同字符数]
            int[][] right = new int[n][3];

            // 从左到右遍历，计算每个位置的左边信息
            int num = 0, mask = 0, count = 0;
            for (int i = 0; i < n - 1; i++) {
                int binary = 1 << (s.charAt(i) - 'a');
                // 如果当前字符不在当前分区的字符集合中
                if ((mask & binary) == 0) {
                    count++;
                    // 如果当前分区不同字符数不超过k，将该字符加入当前分区
                    if (count <= k) {
                        mask |= binary;
                    } else {
                        // 否则开始新分区
                        num++;
                        mask = binary;
                        count = 1;
                    }
                }
                left[i + 1][0] = num;
                left[i + 1][1] = mask;
                left[i + 1][2] = count;
            }

            // 从右到左遍历，计算每个位置的右边信息
            num = 0;
            mask = 0;
            count = 0;
            for (int i = n - 1; i > 0; i--) {
                int binary = 1 << (s.charAt(i) - 'a');
                // 如果当前字符不在当前分区的字符集合中
                if ((mask & binary) == 0) {
                    count++;
                    // 如果当前分区不同字符数不超过k，将该字符加入当前分区
                    if (count <= k) {
                        mask |= binary;
                    } else {
                        // 否则开始新分区
                        num++;
                        mask = binary;
                        count = 1;
                    }
                }
                right[i - 1][0] = num;
                right[i - 1][1] = mask;
                right[i - 1][2] = count;
            }

            // 计算在每个位置进行字符修改后的最大分区数
            int maxVal = 0;
            for (int i = 0; i < n; i++) {
                // 基本分区数：左边分区数 + 右边分区数 + 2（左右各一个新分区）
                int seg = left[i][0] + right[i][0] + 2;
                // 合并左右字符掩码，得到总的不同字符集合
                int totMask = left[i][1] | right[i][1];
                // 计算总的不同字符数
                int totCount = Integer.bitCount(totMask);
                
                // 如果左右分区都已经有k个不同字符，且总不同字符数小于26，可以增加一个分区, 即i位置变成新的字符,单独一个新分区
                if (left[i][2] == k && right[i][2] == k && totCount < 26) {
                    seg++;
                } 
                // 如果总不同字符数加1不超过k，需要减少一个分区,即左右分区合并为一个分区
                else if (Math.min(totCount + 1, 26) <= k) {
                    seg--;
                }
                maxVal = Math.max(maxVal, seg);
            }
            return maxVal;
        }
    }
```
