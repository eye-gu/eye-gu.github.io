```java
class Solution {
        public long countMajoritySubarrays(int[] nums, int target) {
            int preSum = 0; // 数组转换为等于target为1,不等于target为-1后,计算的前缀和
            int lessPreSumCount = 0; // 小于当前n的前缀和的个数,因为p[i] < p[j]时,说明i到j的子数组中,target的个数大于不等于target的个数
            long ans = 0;
            Map<Integer, Integer> cnt = new HashMap<>();//前缀和 -> 次数
            cnt.put(0, 1); // 前缀和0的次数是1个
            for (int n : nums) {
                if (n == target) {
                    // 等于target,要根据当前preSum先计算小于当前preSum的个数
                    lessPreSumCount += cnt.getOrDefault(preSum, 0);
                    preSum++;
                } else {
                    // 不等于target,要先更新前缀和,再计算小于当前preSum的个数
                    // 因为当前lessPreSumCount是前缀和中[-max,preSum)的个数,不等于target,当前位置的前缀和变成了preSum-1,小于当前preSum的范围变成了[-max,preSum-1)
                    preSum--;
                    lessPreSumCount -= cnt.getOrDefault(preSum, 0);
                }
                ans += lessPreSumCount;
                cnt.put(preSum, cnt.getOrDefault(preSum, 0) + 1);
            }
            return ans;
        }
    }
```
