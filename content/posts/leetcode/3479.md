https://leetcode.cn/problems/fruits-into-baskets-iii/description/

官解第一种分块就先不写了, 第二种线段树照着写一下

```go
func numOfUnplacedFruits(fruits []int, baskets []int) int {
	n := len(fruits)
	segTree := make([]int, 4*n)
	for i := 0; i < 4*n; i++ {
		segTree[i] = -1
	}
	buildSegmentTree(baskets, segTree, 1, 0, n-1)
	unplaced := 0
	for _, fruit := range fruits {
		l, r := 0, n-1
		res := -1
		for l <= r {
			mid := (l + r) / 2
			if query(segTree, 1, 0, n-1, 0, mid) >= fruit {
        // 在0-mid的最大值大于等于fruit
				res = mid
				r = mid - 1
			} else {
        // 否则在mid+1-n区间找
				l = mid + 1
			}
		}
		if res != -1 && baskets[res] >= fruit {
			// 找到一个桶可以放下这个水果
			update(segTree, 1, 0, n-1, res, -1)
		} else {
			// 没有找到合适的桶，水果无法放置
			unplaced++
		}
	}
	return unplaced
}

// 构建区间树
func buildSegmentTree(baskets []int, segTree []int, p int, l int, r int) {
	// 叶子结点存储桶的大小
	if l == r {
		segTree[p] = baskets[l]
		return
	}
	mid := (l + r) / 2
	buildSegmentTree(baskets, segTree, 2*p, l, mid)
	buildSegmentTree(baskets, segTree, 2*p+1, mid+1, r)
	// 非叶子结点存储区间的最大值
	segTree[p] = max(segTree[2*p], segTree[2*p+1])
}

// 修改区间树，将位置 i 的值更新为 v
func update(segTree []int, p int, l int, r int, i int, v int) {
	if l == r {
		segTree[p] = v
		return
	}
	mid := (l + r) / 2
	if i <= mid {
		update(segTree, 2*p, l, mid, i, v)
	} else {
		update(segTree, 2*p+1, mid+1, r, i, v)
	}
	segTree[p] = max(segTree[2*p], segTree[2*p+1])
}

// 查询区间 [ql, qr] 的最大值
func query(segTree []int, p int, l int, r int, ql int, qr int) int {
	if ql > r || qr < l {
		return -1
	}
	if ql <= l && qr >= r {
		return segTree[p]
	}
	mid := (l + r) / 2
	// 分别查询左子树和右子树
	return max(query(segTree, 2*p, l, mid, ql, qr), query(segTree, 2*p+1, mid+1, r, ql, qr))
}
```
