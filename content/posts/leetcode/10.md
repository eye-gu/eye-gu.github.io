impl Solution {
  pub fn is_match(s: String, p: String) -> bool {
    let s = s.chars().collect::<Vec<_>>();
    let p = p.chars().collect::<Vec<_>>();
    // dp[i][j] 表示 s前i个字符 是否匹配 p前j个字符. 注意前i个字符, 索引中是i-1
    let mut dp = vec![vec![false; p.len() + 1]; s.len() + 1];
    // 两个空串匹配
    dp[0][0] = true;
    for i in 0..=s.len() {
      for j in 1..=p.len() {
        if p[j - 1] == '*' {
          // 题目保证*前必有字符，j >= 1，因此可安全访问 p[j - 2]
          // *匹配0次的情况, 举例: p为abc*, s为ab, dp[i][j - 2]代表ab和ab匹配
          let no_match = dp[i][j - 2];
          // *匹配1次或多次的情况
          // 举例: p为abc*, s为abc, 此时p[j - 2]为c, s[i - 1]也为c, 说明可以用*匹配, 最后dp[i - 1][j]表示ab和abc*匹配
          let can_repeat = i > 0 && (s[i - 1] == p[j - 2] || p[j - 2] == '.');
          dp[i][j] = no_match || (can_repeat && dp[i - 1][j]);
        } else {
          // dp[i - 1][j - 1] 表示 s[i - 1] 匹配 p[j - 1]
          // (s[i - 1] == p[j - 1] || p[j - 1] == '.') 表示s[i-1]和p[i-1]相等, 或者 p[i-1]是., 可以匹配任意字符
          dp[i][j] = i > 0 && dp[i - 1][j - 1] && (s[i - 1] == p[j - 1] || p[j - 1] == '.');
        }
      }
    }
    dp[s.len()][p.len()]
  }
}
