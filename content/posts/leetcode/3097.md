https://leetcode.cn/problems/shortest-subarray-with-or-at-least-k-ii/description/



```python
class Solution:
    def minimumSubarrayLength(self, nums: List[int], k: int) -> int:
        if k == 0:
            return 1
        # bit_count[i]代表i位置上的1的数量
        bit_count = [0] * 32
        left = 0
        right = 0
        or_num = 0
        ans = len(nums) + 1
        while left < len(nums):
            while right < len(nums) and or_num < k:
                num = nums[right]
                or_num = or_num | num
                i = 0
                while num > 0:
                    if num & 1 == 1:
                        bit_count[i] += 1
                    num >>= 1
                    i = i + 1
                right += 1
                
            if or_num >= k:
                ans = min(ans, right - left)

            num = nums[left]
            i = 0
            while num > 0:
                if num & 1 == 1:
                    bit_count[i] -= 1
                    # 如果某个位置上没有1了, 那么总的or的结果, 就需要减去. 通过1<<i计算该位置上对应的数字
                    if bit_count[i] == 0:
                        or_num -= (1 << i)
                num >>= 1
                i = i + 1

            left += 1
        return -1 if ans == len(nums) + 1 else ans
```
