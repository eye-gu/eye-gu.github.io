```python
import numpy as np
import tensorflow as tf
```


```python
w = tf.Variable(0, dtype=tf.float32)
optimizer = tf.keras.optimizers.Adam(0.1)

def train_step():
    with tf.GradientTape() as tape:
        cost = w ** 2 - 10 * w + 25
    trainable_bariables = [w]
    grads = tape.gradient(cost, trainable_bariables)
    optimizer.apply_gradients(zip(grads, trainable_bariables))
```


```python
print(w)
for i in range(1000):
    train_step()
print(w)
```

```python
w = tf.Variable(0, dtype=tf.float32)
x = np.array([1, -10, 25], dtype=float)
optimizer = tf.keras.optimizers.Adam(0.1)

def training(x, w, optimizer):
    def cost_fn():
        return x[0] * w ** 2 + x[1] * w + x[2]
    for i in range(1000):
        optimizer.minimize(cost_fn, [w])
    return w

w = training(x, w, optimizer)
print(w)
```
